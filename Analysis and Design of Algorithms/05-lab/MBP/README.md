```go
func bottleNeckPath(g *Graph, distances [][]int) [][]int {
	for k := 1; k <= g.V; k++ {
		for i := 1; i <= g.V; i++ {
			for j := 1; j <= g.V; j++ {
				tmp := int(math.Min(float64(distances[i][j]), math.Max(float64(distances[i][k]), float64(distances[k][j]))))
				distances[i][j] = tmp
				distances[j][i] = tmp
			}
		}
	}
	return distances
}
```
